	\section{ارتباط با نظریه اتوماتا و زبان‌ها}
	
	\begin{itemize}
		\item یک برنامه در حال اجرا را می‌توان به عنوان یک \lr{State Machine} در نظر گرفت - هر وضعیت حافظه یک \lr{State} است و هر دستور، یک \lr{Transition}.
		\item خواصی که روی برنامه بررسی می‌شوند، با استفاده از \lr{Linear Temporal Logic (LTL)} بیان می‌شوند.
		\item هر فرمول \lr{LTL} معادل یک \lr{Büchi Automaton} است - یک ماشین حالت محدود که روی کلمات نامتناهی کار می‌کند.
		\item ابزار \lr{SPIN} این اتوماتا را با مدل سیستم ترکیب کرده و تمام حالت‌های ممکن را جستجو می‌کند.
	\end{itemize}
	
	پس آنچه در این پروژه انجام می‌دهیم، عملاً همان نظریه‌ای است که در کلاس خوانده‌ایم را به صورت عملی و قابل لمس پیاده‌سازی می‌کنیم.

	\chapter{توضیح کلاس Main}
	
	کلاس \lr{Main} نقطه شروع اجرای برنامه است و وظیفه هماهنگی بین بخش‌های مختلف را بر عهده دارد.
	
	\section{خواندن ورودی}
	
	\setLTR
	\begin{lstlisting}[language=Java]
		CharStream input = CharStreams.fromPath(Paths.get(args[0]));
	\end{lstlisting}
	\setRTL
	
	این خط فایل ورودی که به عنوان آرگومان خط فرمان دریافت شده است را به جریان کاراکتری تبدیل می‌کند. این جریان سپس توسط \lr{Lexer} پردازش می‌شود. استفاده از \lr{Paths.get} و \lr{CharStreams.fromPath} روش استاندارد و امن برای خواندن فایل در \lr{JAVA} است.
	
	\section{تولید Lexer و Parser}
	
	\setLTR
	\begin{lstlisting}[language=Java]
		HashLexer lexer = new HashLexer(input);
		CommonTokenStream tokens = new CommonTokenStream(lexer);
		HashParser parser = new HashParser(tokens);
	\end{lstlisting}
	\setRTL
	
	\begin{itemize}
		\item $Lexer$: جریان کاراکتری را به توکن‌های معنادار (مانند کلیدواژه‌ها، عملگرها، شناسه‌ها و اعداد) تبدیل می‌کند.
		\item $Token Stream$: جریان توکن‌ها را به صورتی که \lr{Parser} بتواند از آن استفاده کند، سازماندهی می‌کند.
		\item $Parser$: بر اساس گرامر تعریف شده در فایل \lr{Hash.g4}، توکن‌ها را تحلیل کرده و درخت نحوی انتزاعی تولید می‌کند.
	\end{itemize}
	
	\section{بررسی خطاهای نحوی}
	
	\setLTR
	\begin{lstlisting}[language=Java]
		if (parser.getNumberOfSyntaxErrors() > 0) {
			System.out.println("Syntax errors found. Semantic checking skipped.");
			return;
		}
	\end{lstlisting}
	\setRTL
	
	در صورت وجود خطای نحوی، ادامه پردازش متوقف می‌شود. این کار از تولید خروجی ناقص یا اشتباه جلوگیری می‌کند. خطاهای نحوی ممکن است شامل استفاده از کلیدواژه‌های اشتباه، قرارگیری نادرست پرانتزها، یا ساختارهای غیرمجاز باشند.
	
	\section{بررسی معنایی}
	
	\setLTR
	\begin{lstlisting}[language=Java]
		SemanticChecker checker = new SemanticChecker();
		checker.collectDef(tree);
		checker.SemanticCheck(tree);
	\end{lstlisting}
	\setRTL
	
	در این مرحله، دو عملیات مهم انجام می‌شود:
	
	\begin{enumerate}
		\item \textbf{جمع‌آوری تعاریف}: \lr{collectDef} تمام تعاریف متغیرها، توابع و کلاس‌ها را از درخت استخراج کرده و در یک جدول نماد (\lr{Symbol Table}) ذخیره می‌کند.
		\item \textbf{بررسی معنایی}: \lr{SemanticCheck} قواعد زیر را بررسی می‌کند:
		\begin{itemize}
			\item استفاده از متغیرها قبل از تعریف
			\item تطابق نوع داده در عملیات‌ها
			\item تعریف تکراری متغیرها
			\item استفاده صحیح از ساختارهای شرطی و حلقه‌ها
			\item و ....
		\end{itemize}
	\end{enumerate}
	
	اگر خطای معنایی وجود داشته باشد، پیام‌های خطا نمایش داده می‌شوند و برنامه ادامه نمی‌یابد.
	
	\section{تولید کد Promela}
	
	\setLTR
	\begin{lstlisting}[language=Java]
		HashToPromela translator = new HashToPromela();
		String outputCode = translator.visit(tree);
	\end{lstlisting}
	\setRTL
	
	در این مرحله، درخت نحوی توسط \lr{Visitor} پیمایش می‌شود و برای هر گره، کد معادل \lr{Promela} تولید می‌گردد. استفاده از \lr{Visitor} به ما امکان می‌دهد تا دقیقاً کنترل کنیم که برای هر نوع گره چه کدی تولید شود.
	
	\section{ذخیره خروجی}
	
	\setLTR
	\begin{lstlisting}[language=Java]
		Path outputDir = Paths.get("src/codes");
		Path outputFile = outputDir.resolve("output.pml");
		Files.writeString(outputFile, outputCode);
	\end{lstlisting}
	\setRTL
	
	کد تولید شده در فایل \lr{output.pml} در پوشه \lr{src/codes} ذخیره می‌شود. این مسیر استاندارد باعث می‌شود که خروجی به راحتی قابل دسترسی و استفاده در فازهای بعدی باشد.
	
	\chapter{چرا از Visitor استفاده شده است؟}
	
	\section{مقایسه Visitor و Listener}
	
	در این پروژه از الگوی \lr{Visitor} به جای \lr{Listener} استفاده شده است. جدول زیر مقایسه جامعی بین این دو الگو ارائه می‌دهد:
	
	\begin{table}[h!]
		\centering
		\begin{tabular}{|p{4cm}|p{5cm}|p{5cm}|}
			\hline
			\textbf{ویژگی} & \textbf{Visitor} & \textbf{Listener} \\
			\hline
			مقدار بازگشتی & دارد (می‌تواند هر نوع داده‌ای برگرداند) & ندارد (void) \\
			کنترل پیمایش & کامل (خودتان تصمیم می‌گیرید چه گره‌هایی را ببینید) & محدود (پیمایش خودکار توسط ANTLR) \\
			کاربرد اصلی & تولید کد، تبدیل، تحلیل عمیق & تحلیل ساده، جمع‌آوری اطلاعات \\
			وابستگی به گرامر & کم (هر گره یک متد مجزا دارد) & زیاد (همه در یک متد) \\
			پیچیدگی پیاده‌سازی & بیشتر (نیاز به مدیریت پیمایش) & کمتر (پیمایش خودکار) \\
			قابلیت توسعه & بالا (افزودن عملیات جدید بدون تغییر گره‌ها) & متوسط \\
			\hline
		\end{tabular}
		\caption{مقایسه الگوهای Visitor و Listener}
	\end{table}
	
	\section{دلایل اصلی انتخاب Visitor}
	
	\begin{enumerate}
		\item \textbf{تولید مقدار بازگشتی}: مهم‌ترین دلیل، نیاز به تولید کد \lr{Promela} به عنوان خروجی است. در \lr{Visitor}، هر متد می‌تواند یک رشته (یا هر نوع داده دیگری) برگرداند، در حالی که \lr{Listener} خروجی ندارد.
		
		\item \textbf{کنترل کامل بر پیمایش}: در \lr{Visitor}، خودمان تصمیم می‌گیریم که چه گره‌هایی را بازدید کنیم و به چه ترتیبی. این ویژگی برای ترجمه دقیق دستورات \lr{Hash} به \lr{Promela} ضروری است، زیرا ترتیب اجرای دستورات و ساختارهای کنترلی باید به درستی حفظ شود.
		
		\item \textbf{بهبود قابلیت نگهداری}: با استفاده از \lr{Visitor}، هر قانون گرامر یک متد مجزا دارد که کار اصلاح و توسعه را بسیار ساده‌تر می‌کند. اگر بخواهیم یک دستور جدید اضافه کنیم، فقط یک متد جدید می‌نویسیم.
		
		\item \textbf{مدیریت بهتر Context}: در فرآیند ترجمه، نیاز به نگهداری وضعیت‌هایی مانند برچسب‌های حلقه‌ها یا پرچم‌های استثنا داریم. \lr{Visitor} با استفاده از متغیرهای عضو کلاس یا پشته‌ها، این کار را به سادگی امکان‌پذیر می‌کند.
		
		\item \textbf{انتخابی بازدید از گره‌ها}: در \lr{Visitor}، می‌توانیم از بازدید برخی گره‌ها صرف‌نظر کنیم یا گره‌های خاصی را چندین بار بازدید کنیم، در حالی که در \lr{Listener} همه گره‌ها دقیقاً یک بار بازدید می‌شوند.
	\end{enumerate}
	
	\section{نمونه‌های عملی از مزایای Visitor}
	
	\begin{itemize}
		\item \textbf{مدیریت تقسیم بر صفر}: در متدهای مختلف (مانند \lr{visitWhileStmt})، می‌توانیم تصمیم بگیریم که قبل از شرط اصلی حلقه، یک شرط اضافی برای بررسی تقسیم بر صفر اضافه کنیم.
		
		\item \textbf{مدیریت حلقه‌های تو در تو}: با استفاده از پشته  \lr{continueLabels}، می‌توانیم به راحتی برچسب‌های حلقه‌های تو در تو را مدیریت کنیم. هر بار که وارد یک حلقه جدید می‌شویم، برچسب آن را به پشته اضافه می‌کنیم و پس از خروج، آن را حذف می‌کنیم.
		
		\item \textbf{مدیریت استثناها}: در ساختار \lr{try-catch}، با استفاده از پرچم‌های استثنا که روی پشته قرار می‌گیرند، می‌توانیم رفتار استثنا را در \lr{Promela} شبیه‌سازی کنیم.
	\end{itemize}
	
	\chapter{طراحی Translator (HashToPromela)}
	
	کلاس \lr{HashToPromela} مسئول اصلی تبدیل کد \lr{Hash} به \lr{Promela} است. \newline این کلاس از \lr{HashBaseVisitor<String>} ارث‌بری کرده و برای هر گره از درخت نحوی، یک رشته خروجی تولید می‌کند.
	
	\section{ساختار کلی کلاس}
	
	\begin{itemize}
		\item \textbf{متغیرهای سراسری اضافه شده}: 
		\begin{itemize}
			\item \lr{divByZero}: برای بررسی ویژگی Safety (جلوگیری از تقسیم بر صفر)
			\item \lr{inLoop} و \lr{exitLoop}: برای بررسی ویژگی‌های Liveness
			\item \lr{endReached}: برای بررسی ویژگی Reachability
			\item \lr{activeLoopCount}: برای مدیریت حلقه‌های تو در تو
		\end{itemize}
		
		\item \textbf{پشته‌ها}:
		\begin{itemize}
			\item \lr{continueLabels}: ذخیره برچسب‌های حلقه‌ها برای دستور \lr{edame}
			\item \lr{exceptionFlags}: ذخیره پرچم‌های استثنا برای ساختار \lr{try-catch}
			\item \lr{catchLabels}: ذخیره برچسب‌های بلوک‌های \lr{catch}
		\end{itemize}
		
		\item \textbf{شمارنده‌ها}:
		\begin{itemize}
			\item \lr{loopCounter}: تولید برچسب‌های یکتا برای حلقه‌ها
			\item \lr{exceptionCounter}: تولید شناسه‌های یکتا برای استثناها
		\end{itemize}
	\end{itemize}
	
	\section{متد visitProgram - نقطه ورود اصلی}
	
	این متد مهم‌ترین متد کلاس است و ساختار کلی فایل خروجی را تولید می‌کند:
	
	\begin{enumerate}
		\item \textbf{تعریف متغیرهای کمکی}: در ابتدای خروجی، تمام متغیرهای کمکی که برای بررسی خواص سیستم نیاز هستند، تعریف می‌شوند:
		\setLTR
		\begin{lstlisting}[language=Promela]
			bool divByZero = false;
			bool inLoop = false;
			bool exitLoop = false;
			bool endReached = false;
			int activeLoopCount = 0;
		\end{lstlisting}
		\setRTL
		
		\item \textbf{پردازش متغیرهای سراسری}: اعلان‌های سطح بالا که از نوع تعریف متغیر هستند، شناسایی شده و به صورت متغیرهای سراسری در \lr{Promela} تعریف می‌شوند. این کار باعث می‌شود که متغیرها در کل برنامه قابل دسترسی باشند.
		
		\item \textbf{تعریف بدنه اصلی}: سایر گره‌های سطح بالا (دستورات، توابع، کلاس‌ها) درون یک پروسه فعال با نام \lr{main} قرار می‌گیرند ، همچنین متغیر های سراسری در این قسمت مقدار دهی میشوند:
		\setLTR
		\begin{lstlisting}[language=Promela]
			active proctype main() {
				\\ body
			}
		\end{lstlisting}
		\setRTL
		
		\item \textbf{افزودن برچسب پایان}:\newline در انتهای بدنه، برچسب \lr{endReachedLabel} و دستور تنظیم \lr{endReached = true} اضافه می‌شود تا بتوان ویژگی \lr{Reachability} را بررسی کرد.
	\end{enumerate}

		\item سپس \lr{visit(ctx.multiplicativeExpr(1))} را صدا می‌زند که به \lr{"b"} می‌رسد
		\item در نهایت \lr{"a + b"} را برمی‌گرداند
	\end{enumerate}

	\subsection{مدیریت عملگرهای خاص}
	
	\begin{itemize}
		\item \textbf{توان (\lr{**})}: با تابع \lr{expandPower} به ضرب متوالی تبدیل می‌شود:
		\begin{itemize}
			\item \lr{x**3} $\rightarrow$ \lr{x * x * x}
		\end{itemize}
		
		\item \textbf{افزایش/کاهش (\lr{++}, \lr{--})}: ترتیب ارزیابی پیشوند و پسوند حفظ می‌شود:
		\begin{itemize}
			\item \lr{x++} $\rightarrow$ ابتدا مقدار فعلی، سپس افزایش
			\item \lr{++x} $\rightarrow$ ابتدا افزایش، سپس مقدار جدید
		\end{itemize}
		
		\item \textbf{تقسیم بر صفر}: با تابع \lr{guardDivisionByZero} مدیریت می‌شود
	\end{itemize}
	
	برای بررسی صحت ترجمه عملگر توان و عملگرهای افزایش، تست هشتم با نام \lr{test08\_power\_incdec.hash}طراحی شد. 
	هدف این تست آن است که نشان دهد مترجم می‌تواند عملگر \lr{**} را، که در \lr{Promela} به صورت مستقیم وجود ندارد، به ضرب متوالی تبدیل کند. 
	همچنین این تست بررسی می‌کند که عملگرهای \lr{++} به assignment معادل در \lr{Promela} تبدیل می‌شوند.
	
	
	در این برنامه، مقدار اولیه متغیر \lr{x} برابر ۲ است. 
	سپس مقدار \lr{y} با استفاده از عبارت \lr{x ** 3} محاسبه می‌شود. 
	از آنجا که \lr{Promela} عملگر توان ندارد، انتظار داریم این عبارت به صورت ضرب متوالی \lr{x * x * x} تولید شود. 
	در ادامه، عملگرهای \lr{x++} و \lr{++y} نیز استفاده شده‌اند و باید به افزایش یک‌واحدی متغیرها تبدیل شوند.
	
	کد ورودی \lr{Hash} برای این تست به صورت زیر است:
	
	\begin{latin}

			adad y = 0;
			
			y = x ** 3;
			x++;
			++y;
			
			bechap(y);
		\end{lstlisting}
	\end{latin}

			}

	در این خروجی، متغیرهای \lr{x} و \lr{y} به صورت سراسری تعریف شده‌اند و مقداردهی اولیه آن‌ها داخل \lr{active proctype main} انجام شده است. 
	نکته اصلی این تست در ترجمه عبارت \lr{x ** 3} دیده می‌شود. 
	چون \lr{Promela} عملگر توان ندارد، مترجم این عبارت را به ضرب متوالی تبدیل کرده است و خروجی \lr{((x) * (x) * (x))} تولید شده است. 
	پرانتزهای اطراف \lr{x} باعث می‌شوند اگر پایه توان یک expression پیچیده‌تر باشد، ترتیب ارزیابی و تقدم عملگرها حفظ شود.
	
	همچنین دستور \lr{x++} در خروجی به \lr{x = x + 1} تبدیل شده است. 
	این تبدیل نشان می‌دهد که عملگر افزایش پسوندی در حالتی که به صورت یک statement مستقل استفاده شود، معادل افزایش یک‌واحدی متغیر است. 
	به همین صورت، دستور \lr{++y} نیز به \lr{y = y + 1} تبدیل شده است. 
	در این تست، چون هر دو عملگر به صورت statement مستقل آمده‌اند، خروجی نهایی هر دو به شکل assignment ساده تولید شده است.
	
	این تست نشان می‌دهد که مترجم می‌تواند عملگر توان با توان ثابت را به ضرب متوالی تبدیل کند و همچنین عملگرهای افزایش را به assignmentهای معادل در \lr{Promela} تبدیل نماید. 
	بنابراین بخش مربوط به عملگرهای خاص در \lr{HashToPromela}، یعنی تبدیل \lr{**} و \lr{++}، در این نمونه به درستی عمل کرده است.

	\section{متد visitIfStmt - ترجمه دستور شرطی}
	
	دستور شرطی \lr{age/bood/vagarna} به ساختار \lr{if :: ... :: else -> ... fi} در \lr{Promela} تبدیل می‌شود:
	\setLTR
	\begin{lstlisting}[language=Promela]
		if
		:: (condition) -> 
		// then block
		:: else ->
		// else block (optional)
		fi
	\end{lstlisting}
	\setRTL
	
	همچنین، اگر شرط شامل عملیات تقسیم باشد، یک شرط اضافی برای بررسی تقسیم بر صفر قبل از شرط اصلی اضافه می‌شود.
	
	\noindent
	برای بررسی صحت ترجمه دستور شرطی، تستی با نام \lr{test02\_if\_else.hash} در نظر گرفته شد.
	هدف این تست آن است که نشان دهد ساختار شرطی \lr{age/bood/vagarna} در زبان \lr{Hash} به درستی به ساختار شرطی \lr{if/fi} در \lr{Promela} تبدیل می‌شود.
	در این مثال، مقدار متغیر \lr{x} برابر ۳ است و شرط \lr{x > 0} بررسی می‌شود.
	اگر شرط برقرار باشد، مقدار \lr{y} برابر ۱ می‌شود و در غیر این صورت مقدار آن برابر ۲ قرار می‌گیرد.
	
	کد ورودی \lr{Hash} برای این تست به صورت زیر است:
	
	\begin{latin}
		\begin{lstlisting}[style=hashstyle]
			adad x = 3;
			adad y = 0;
			
			age (x > 0) bood {
				y = 1;
			} vagarna {
				y = 2;
			}
			
			bechap(y);
		\end{lstlisting}
	\end{latin}
	
	خروجی تولیدشده توسط مترجم در فایل \lr{output.pml} به صورت زیر است:
	
	\begin{latin}
		\begin{lstlisting}[style=codestyle]
			bool divByZero = false;
			bool inLoop = false;
			bool exitLoop = false;
			bool endReached = false;
			int activeLoopCount = 0;
			
			int x;
			int y;
			
			active proctype main() {
				x = 3;
				y = 0;
				if
				:: (x > 0) ->
				y = 1;
				:: else ->
				y = 2;
				fi
				printf("%d\n", y);
				endReachedLabel:
				endReached = true;
				skip;
			}
		\end{lstlisting}
	\end{latin}
	
	در این خروجی، نوع \lr{adad} برای متغیرهای \lr{x} و \lr{y} به نوع \lr{int} در \lr{Promela} تبدیل شده است.
	همچنین مقداردهی اولیه متغیرها داخل \lr{active proctype main} انجام شده است.

	در این خروجی، متغیرهای \lr{x}، \lr{y} و \lr{safe} به صورت سراسری تعریف شده‌اند و مقداردهی اولیه آن‌ها داخل \lr{active proctype main} انجام شده است. 
	پس از مقداردهی اولیه، متغیر محلی \lr{errFlag\_1} با مقدار \lr{false} ساخته شده است. 
	این متغیر نشان می‌دهد که آیا در زمان اجرای بلوک \lr{emtehan} خطایی رخ داده است یا نه.
	
	نکته اصلی این تست در ترجمه دستور \lr{safe = x / y} دیده می‌شود. 
	چون این دستور شامل تقسیم است، مترجم ابتدا شرط \lr{y == 0} را تولید کرده است. 
	اگر این شرط برقرار باشد، مقدار \lr{divByZero} برابر \lr{true} می‌شود تا وقوع تقسیم بر صفر برای فاز دوم قابل بررسی باشد. 
	همچنین مقدار \lr{errFlag\_1} برابر \lr{true} قرار می‌گیرد و سپس با دستور \lr{goto catchCheck\_1} کنترل برنامه به بخش مربوط به \lr{catch} منتقل می‌شود. 
	در این حالت، دستور اصلی تقسیم اجرا نمی‌شود.
	
	اگر مقدار \lr{y} صفر نباشد، شاخه \lr{else} اجرا می‌شود و دستور اصلی \lr{safe = x / y} انجام می‌گیرد. 
	پس از آن، برنامه به برچسب \lr{catchCheck\_1} می‌رسد. 
	در این قسمت، اگر \lr{errFlag\_1} فعال شده باشد، بدنه \lr{gereftar} اجرا می‌شود و مقدار \lr{safe} برابر صفر قرار می‌گیرد. 
	اگر خطایی رخ نداده باشد، شاخه \lr{else -> skip} اجرا می‌شود و برنامه بدون اجرای بخش catch ادامه پیدا می‌کند.
	
	این تست نشان می‌دهد که مترجم ساختار \lr{emtehan/gereftar} را با استفاده از \lr{errFlag}، برچسب \lr{catchCheck} و دستور \lr{goto} به درستی در \lr{Promela} شبیه‌سازی می‌کند. 
	همچنین نشان می‌دهد که اگر تقسیم بر صفر داخل بلوک \lr{emtehan} رخ دهد، علاوه بر فعال شدن \lr{divByZero}، کنترل برنامه به بخش \lr{gereftar} منتقل می‌شود. 
	این رفتار با طراحی کلی مترجم سازگار است، زیرا \lr{Promela} exception واقعی ندارد و این سازوکار جایگزین برای مدل کردن try-catch در خروجی است.

	همچنین چون این تقسیم خارج از ساختار \lr{emtehan/gereftar} قرار دارد، در خروجی فقط متغیر \lr{divByZero} فعال می‌شود و خبری از \lr{errFlag} یا پرش به \lr{catchCheck} نیست.

	این خروجی برای فاز دوم نیز مهم است، چون ویژگی \lr{Safety} با استفاده از متغیر \lr{divByZero} بررسی می‌شود.

		\item \textbf{سناریوی نقض:} روی یک برنامه‌ی طراحی‌شده که مسیر بحرانی در آن واقعاً \lr{reachable} است، تا نشان داده شود \lr{SPIN} می‌تواند نقض واقعی را هم تشخیص دهد (نه اینکه صرفاً همیشه \lr{pass} بدهد).
	\end{itemize}
	خصوصیت \lr{Deadlock Freedom} از این قاعده مستثناست؛ دلیل آن در بخش مربوطه توضیح داده شده است.

	\subsection{چرا فرمول به این شکل است؟}
	
	خصوصیت‌های \lr{Liveness} معمولاً به صورت
	
	\[
	\square(P \rightarrow \Diamond Q)
	\]
	
	بیان می‌شوند. مفهوم این فرمول آن است که هرگاه شرط $P$ برقرار شود، سیستم موظف است در ادامه اجرای خود نهایتاً به حالتی برسد که $Q$ برقرار باشد.

	\[
	\square(\mathit{inLoop}\rightarrow\Diamond\mathit{exitLoop})
	\]
	
	استفاده شده است.
	
	اگر تنها از عملگر $\Diamond$ استفاده می‌شد، مانند
	
	\[
	\Diamond(\mathit{exitLoop})
	\]

	\subsection{فرمول LTL}
	فرمولی که به \lr{SPIN} داده می‌شود، نقیض فرمول اصلی است:
	\ltlbox{\lnot(\Diamond\, \mathit{endReached})}
	معادل آن در \lr{properties.ltl}: \lr{ltl reachEnd: ! (<> (endReached))}
	
	\textbf{نکته‌ی تفسیری مهم:} چون فرمول به‌صورت نقیض داده می‌شود، برخلاف بقیه‌ی خصوصیات، در این مورد \textbf{یافتن یک نقض (\lr{errors: 1})} به معنای \textbf{تایید} \lr{reachability} است (یعنی مسیری به \lr{endReached} واقعاً وجود دارد)، و \textbf{عدم یافتن نقض (\lr{errors: 0})} به معنای آن است که \lr{endReached} در آن مدل خاص \textbf{غیرقابل‌دسترس} است.
	
	\subsection{چرا فرمول به این شکل است؟}
	
	در SPIN معمولاً خصوصیت‌های دسترس‌پذیری به صورت مستقیم بیان نمی‌شوند، زیرا LTL ذاتاً برای بیان خواصی که باید در تمام مسیرهای اجرا برقرار باشند طراحی شده است.
	
	برای بررسی اینکه آیا حالت پایانی برنامه قابل دسترسی است، نقیض این ویژگی نوشته می‌شود:
	
	\[
	\neg\Diamond(\mathit{endReached})
	\]
	
	در صورت نقض این فرمول توسط SPIN، یک مسیر نمونه ارائه می‌شود که در آن متغیر
	\lr{endReached}
	فعال شده است. بنابراین شکست این فرمول به معنای آن است که حالت پایانی برنامه واقعاً قابل دستیابی بوده و ویژگی Reachability برقرار است.

	\section{مقدمه فاز سوم}
	
	در فاز دوم، پنج خصوصیت مشخص را با استفاده از \lr{SPIN} روی چند برنامه‌ی \emph{مشخص و متناهی} \lr{Hash} بررسی کردیم. نتیجه‌ی هر بار اجرای \lr{SPIN} تنها درباره‌ی همان یک برنامه‌ی خاص حرف می‌زد؛ برای مثال، خصوصیت \lr{Invariant} (\lr{nonNegativeX}) فقط ثابت کرد که در \emph{مدل پایه‌ی مشخصِ فاز اول}، متغیر \lr{x} هرگز منفی نمی‌شود --- نه اینکه در \emph{هر} برنامه‌ی \lr{Hash} که از الگوی مشابهی پیروی کند، همین اتفاق می‌افتد.
	
	هدف فاز سوم رفتن از این نتیجه‌ی موردی به یک تضمین کلی و ریاضی است: می‌خواهیم ثابت کنیم که اگر یک شرط \lr{invariant} قبل از شروع یک حلقه برقرار باشد و بدنه‌ی حلقه آن را حفظ کند، آنگاه بعد از پایان کامل حلقه --- برای \textbf{هر} برنامه، \textbf{هر} شرط حلقه، و \textbf{هر} تعداد تکرار --- آن \lr{invariant} همچنان برقرار خواهد بود. 
	
	برای بیان چنین ادعای کلی‌ای، برخلاف فاز دوم که یک مدل متناهی کافی بود، به یک تعریف \textbf{صوری و کامل} از معنای اجرای هر دستور \lr{Hash} نیاز داریم؛ یعنی \lr{Operational Semantics}. سپس ادعا را به عنوان یک قضیه‌ی ریاضی درباره‌ی این تعریف بیان و در \lr{Lean 4} اثبات می‌کنیم.
	\section{اثبات ریاضی قضیه با استقرا روی اشتقاق}
	
	پیش از رفتن سراغ کد \lr{Lean}، ابتدا قضیه را به‌طور کاملاً مستقل و روی کاغذ، با استقرا روی \textbf{ساختار اشتقاق} (derivation) اثبات می‌کنیم. همین منطق در بخش بعد، خط‌به‌خط به کد \lr{Lean} ترجمه می‌شود.
	
	\subsection{بیان دقیق قضیه}
	
	فرض کنید $P$ یک گزاره (\lr{predicate}) روی حالت‌هاست. ادعا می‌کنیم:
	
	\ltlbox{P(\sigma) \;\land\; \bigl(\forall \sigma_1 \sigma_2,\; P(\sigma_1) \land \langle b,\sigma_1 \rangle \Downarrow \text{dorost} \land \langle S,\sigma_1 \rangle \Downarrow \sigma_2 \;\Rightarrow\; P(\sigma_2)\bigr) \;\land\; \bigl(\langle \text{ta}(b)\,\{S\}, \sigma \rangle \Downarrow \sigma' \;\Rightarrow\; P(\sigma')\bigr)}
	
	یعنی: اگر $P$ قبل از حلقه برقرار باشد و بدنه‌ی حلقه آن را حفظ کند ، آنگاه در هر اشتقاقی که $\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma'$ را نتیجه بدهد، $P(\sigma')$ برقرار است.
	
	\subsection{چرا استقرا روی اشتقاق، نه روی عدد تکرار؟}
	
	نکته‌ی مهم این است که در تعریف \lr{Big-Step}، هیچ‌جا عدد تکرار حلقه به‌صراحت شمرده نمی‌شود. آنچه داریم فقط یک \textbf{درخت اشتقاق} است که فرضیه‌ی $\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma'$ را نتیجه می‌دهد. طبق قوانین استنتاجی سند پروژه، تنها \textbf{دو} قانون وجود دارد که نتیجه‌شان می‌تواند به این شکل باشد: \lr{While-F} و \lr{While-T} (چهار قانون دیگر --- \lr{Assign}، \lr{Seq}، \lr{If-T}، \lr{If-F} --- نتیجه‌شان همیشه دستوری غیر از \lr{ta} است و اصلاً نمی‌توانند این حکم را تولید کنند). پس اثبات با \textbf{استقرا روی ساختار همین اشتقاق} انجام می‌شود: دقیقاً دو حالت داریم که باید بررسی شوند --- یکی به‌عنوان پایه‌ی استقرا و دیگری به‌عنوان گام استقرا.
	
	\subsection{پایه‌ی استقرا --- قانون While-F}
	
	فرض کنید آخرین قانون استفاده‌شده در اشتقاق، \lr{While-F} باشد:
	
	\[
	\dfrac{\langle b,\sigma\rangle \Downarrow \mathit{ghalat}}
	{\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma}\;\text{(While-F)}
	\]
	
	طبق این قانون، حالت پایانی دقیقاً همان حالت شروع است، یعنی $\sigma' = \sigma$. پس باید نشان دهیم $P(\sigma')$ برقرار است، که با جای‌گذاری برابر است با نشان‌دادن $P(\sigma)$. اما این دقیقاً همان مقدمه‌ی اول قضیه است که از پیش داریم:
	
	\[
	P(\sigma) \;\Rightarrow\; P(\sigma') \quad (\text{چون } \sigma'=\sigma)
	\]
	
	پس پایه‌ی استقرا بدون نیاز به هیچ استدلال اضافه‌ای برقرار است.
	
	\subsection{گام استقرا --- قانون While-T}
	
	فرض کنید آخرین قانون استفاده‌شده، \lr{While-T} باشد:
	
	\[
	\dfrac{\langle b,\sigma\rangle \Downarrow \mathit{dorost}
		\qquad \langle S,\sigma\rangle \Downarrow \sigma''
		\qquad \langle \mathit{ta}(b)\{S\},\sigma''\rangle \Downarrow \sigma'}
	{\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma'}\;\text{(While-T)}
	\]
	
	این قانون سه مقدمه دارد. سومین مقدمه، یعنی $\langle \mathit{ta}(b)\{S\},\sigma''\rangle \Downarrow \sigma'$، خودش یک اشتقاق کامل و \textbf{مستقل} برای همان حکم $ ta (b) {S} $ است، با این تفاوت که حالت شروعش $\sigma''$ است نه $\sigma$؛ و چون این زیر\nobreakdash-اشتقاق زیرمجموعه‌ای از اشتقاق اصلی است، \textbf{فرض استقرا} (\lr{IH}) برایش صدق می‌کند:
	
	\[
	\text{(IH)}\qquad P(\sigma'') \;\land\; \langle \mathit{ta}(b)\{S\},\sigma''\rangle \Downarrow \sigma' \;\Rightarrow\; P(\sigma')
	\]
	
	پس کافی است نشان دهیم $P(\sigma'')$ برقرار است؛ آنگاه با اعمال \lr{IH}، مستقیماً $P(\sigma')$ نتیجه می‌شود. برای اثبات $P(\sigma'')$، از سه چیزی که در این حالت در دسترس داریم استفاده می‌کنیم:
	
	\begin{itemize}
		\item $P(\sigma)$ (فرض قضیه، چون هنوز روی حالت شروع $\sigma$ هستیم)،
		\item $\langle b,\sigma\rangle \Downarrow \mathit{dorost}$ (اولین مقدمه‌ی \lr{While-T})،
		\item $\langle S,\sigma\rangle \Downarrow \sigma''$ (دومین مقدمه‌ی \lr{While-T})،
	\end{itemize}
	
	که دقیقاً سه شرط مقدمه‌ی دوم قضیه (پایستگی $P$ توسط بدنه) هستند. با اعمال مستقیم آن مقدمه:
	
	\[
	P(\sigma) \;\land\; \langle b,\sigma\rangle{\Downarrow}\mathit{dorost} \;\land\; \langle S,\sigma\rangle{\Downarrow}\sigma'' \;\Rightarrow\; P(\sigma'')
	\]
	
	نتیجه می‌گیریم $P(\sigma'')$. حال با جای‌گذاری در \lr{IH} بالا:
	
	\[
	P(\sigma'') \;\Rightarrow\; P(\sigma')
	\]
	
	و در نهایت $P(\sigma')$ اثبات می‌شود.
	
	\subsection{جمع‌بندی اثبات}
	
	چون هر اشتقاقِ $\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma'$ فقط می‌تواند با یکی از دو قانون \lr{While-F} یا \lr{While-T} ساخته شده باشد، و در \textbf{هر دو} حالت نشان دادیم $P(\sigma')$ برقرار است (در پایه به‌طور مستقیم، در گام با کمک فرض استقرا)، طبق اصل استقرا روی ساختار اشتقاق، قضیه برای \textbf{هر} اشتقاق ممکنی از $\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma'$ درست است. 
	
	\subsection*{پیوند با کد Lean}
	این دقیقاً همان دو شاخه‌ای است که در تابع \lr{aux} بخش بعد، پس از باز کردن \lr{cases hEq}، تحت نام‌های \lr{whileFalse} و \lr{whileTrue} ظاهر می‌شوند: شاخه‌ی \lr{whileFalse} معادل «پایه‌ی استقرا» بالاست (\lr{exact hInv} دقیقاً همان استفاده از $P(\sigma)$ به‌جای $P(\sigma')$ است)، و شاخه‌ی \lr{whileTrue} معادل «گام استقرا» است (\lr{bodyPreservesInvariant} همان اعمال مقدمه‌ی دوم برای رسیدن به $P(\sigma'')$ است، و \lr{ihRest} همان استفاده از فرض استقرا \lr{IH} روی زیر\nobreakdash-اشتقاق کوچکتر است). چهار قانون دیگر (\lr{Assign}، \lr{Seq}، \lr{If-T}، \lr{If-F}) هم در اثبات ریاضی بالا اصلاً ظاهر نشدند، دقیقاً به همان دلیلی که در کد \lr{Lean} با \lr{cases hEq} بلافاصله رد می‌شوند: نتیجه‌ی آن‌ها هرگز از شکل $\langle \mathit{ta}(b)\{S\},\sigma\rangle \Downarrow \sigma'$ نیست.
	
	\section{بازنگری مفاهیم کلیدی}
	
	\subsection{Big-Step}
	در این پروژه از رویکرد \lr{Big-Step} (یا \lr{Natural Semantics}) استفاده شده است. نماد اصلی آن:
	\[
	\langle S, \sigma \rangle \Downarrow \sigma'
	\]
	خوانده می‌شود: «دستور $S$ در محیط $\sigma$ اجرا می‌شود و به محیط $\sigma'$ می‌رسد.» برخلاف \lr{Small-Step} که هر قدم اجرا را جداگانه نشان می‌دهد، \lr{Big-Step} فقط رابطه‌ی بین \emph{حالت شروع} و \emph{حالت پایان کامل اجرا} را بیان می‌کند --- دقیقاً همان چیزی که برای اثبات این قضیه لازم است.
	
	\subsection*{توجه:}
	در فایل \lr{Lean}، \lr{BigStep} به صورت یک \lr{Prop} استقرایی سه‌آرگومانی تعریف شده، نه یک تابع. دلیل این انتخاب این است که اجرای یک حلقه‌ی \lr{ta} (\lr{while}) ممکن است برای برخی ورودی‌ها هرگز خاتمه نیابد؛ یک رابطه به‌سادگی اجازه می‌دهد در چنین حالتی هیچ $\sigma'$ ای با $S$ و $\sigma$ در رابطه نباشد، بدون نیاز به تعریف یک «مقدار خطا» یا «مقدار بی‌نهایت» مصنوعی.
	
	\section{تعریف انواع پایه در Lean}
	
	اولین گام، تعریف صوری نحو (\lr{syntax}) زیرمجموعه‌ی \lr{Hash} است --- دقیقاً همان چیزی که \lr{Parser} فاز اول از روی گرامر \lr{Hash.g4} می‌ساخت، اما این‌ بار به شکل یک نوع داده‌ی \lr{Lean}:
	
	\begin{latin}
		\begin{lstlisting}[style=leanstyle]
			abbrev Var := String
			abbrev State := Var -> Int
			
			inductive Expr where
			| num : Int -> Expr
			| var : Var -> Expr
			| add : Expr -> Expr -> Expr
			| lt : Expr -> Expr -> Expr
			
			inductive Stmt where
			| assign : Var -> Expr -> Stmt
			| seq : Stmt -> Stmt -> Stmt
			| ifElse : Expr -> Stmt -> Stmt -> Stmt
			| while : Expr -> Stmt -> Stmt
		\end{lstlisting}
	\end{latin}
	
	\begin{itemize}
		\item \lr{State} همان $\sigma$ در فرمول‌های ریاضی است: یک تابع از نام متغیر به مقدار صحیح آن. هیچ نوع جداگانه‌ای برای \lr{Bool} تعریف نشده؛ مقادیر بولی زبان \lr{Hash} (\lr{dorost}/\lr{ghalat}) با اعداد صحیح شبیه‌سازی می‌شوند (غیرصفر = درست، صفر = نادرست) --- همان الگویی که در فاز اول برای ترجمه به \lr{Promela} هم استفاده شد.
		\item \lr{Stmt} همان چهار دستور اصلی مثال راهنما را دارد (\lr{assign}، \lr{seq}، \lr{ifElse}، \lr{while}) که برای اثبات این قضیه کافی‌اند؛ ساختارهای \lr{baraye}، \lr{shekan}/\lr{edame} و \lr{gereftar}/\lr{emtehan} طبق صورت پروژه لازم نیست در این فاز صوری‌سازی شوند.
	\end{itemize}
	
	\section{ارزیابی عبارات و به‌روزرسانی State}
	
	\begin{latin}
		\begin{lstlisting}[style=leanstyle]
			def evalExpr (e : Expr) (s : State) : Int :=
			match e with
			| Expr.num n => n
			| Expr.var x => s x
			| Expr.add e1 e2 => evalExpr e1 s + evalExpr e2 s
			| Expr.lt e1 e2 =>
			if evalExpr e1 s < evalExpr e2 s then 1 else 0
			
			def update (s : State) (x : Var) (v : Int) : State :=
			fun y => if y = x then v else s y
		\end{lstlisting}
	\end{latin}
	
	\lr{evalExpr} به‌جای اینکه (مثل قوانین استنتاجی سند پروژه) به شکل یک رابطه‌ی جداگانه با قوانین \lr{Num}، \lr{Var}، \lr{Add} و \ldots تعریف شود، مستقیماً به صورت یک \textbf{تابع بازگشتی} نوشته شده است. برخلاف \lr{BigStep}، ارزیابی عبارات (\lr{Expr}) هیچ‌وقت با مشکل عدم‌خاتمه روبه‌رو نیست — هر عبارت، هرچقدر هم تودرتو، در یک تعداد قدم محدود و ثابت به یک مقدار می‌رسد و هیچ دو مسیر اجرای متفاوتی هم برایش وجود ندارد. به همین دلیل نوشتن \lr{evalExpr} به‌صورت یک تابع معمولی (نه یک رابطه‌ی استقرایی مثل \lr{BigStep}) کاملاً بلامانع است. این تابع همان چیزی را می‌گوید که قوانین استنتاجی سند پروژه (\lr{Num}، \lr{Var}، \lr{Add} و \ldots) می‌گفتند، با یک مزیت اضافه: چون خروجی تابع همیشه به‌خودی‌خود یکتاست (طبق تعریف خودِ تابع در \lr{Lean})، دیگر لازم نیست قطعیت عبارات را جداگانه اثبات کنیم — چیزی که اگر عبارات را هم مثل دستورات با یک رابطه تعریف می‌کردیم، لازم می‌شد (همان گزینه‌ی اول صورت پروژه: \lr{Expression Determinism}).
	
	\lr{update} همان نماد $\sigma[x \mapsto v]$ است: یک تابع (\lr{State}) جدید که با \lr{s} یکسان است مگر روی \lr{x}، که مقدار \lr{v} را می‌گیرد.
	
	\section{رابطه‌ی BigStep}
	
	\begin{latin}
		\begin{lstlisting}[style=leanstyle]
			inductive BigStep : Stmt -> State -> State -> Prop where
			
			| assign {x : Var} {e : Expr} {s : State} :
			BigStep
			(Stmt.assign x e)
			s
			(update s x (evalExpr e s))
			
			| seq {S1 S2 : Stmt} {s sMid sFinal : State} :
			BigStep S1 s sMid ->
			BigStep S2 sMid sFinal ->
			BigStep (Stmt.seq S1 S2) s sFinal
			
			| ifTrue {b : Expr} {S1 S2 : Stmt} {s sFinal : State} :
			evalExpr b s ≠ 0 ->
			BigStep S1 s sFinal ->
			BigStep (Stmt.ifElse b S1 S2) s sFinal
			
			| ifFalse {b : Expr} {S1 S2 : Stmt} {s sFinal : State} :
			evalExpr b s = 0 ->
			BigStep S2 s sFinal ->
			BigStep (Stmt.ifElse b S1 S2) s sFinal
			
			| whileFalse {b : Expr} {body : Stmt} {s : State} :
			evalExpr b s = 0 ->
			BigStep (Stmt.while b body) s s
			
			| whileTrue {b : Expr} {body : Stmt} {s sMid sFinal : State} :
			evalExpr b s ≠ 0 ->
			BigStep body s sMid ->
			BigStep (Stmt.while b body) sMid sFinal ->
			BigStep (Stmt.while b body) s sFinal
		\end{lstlisting}
	\end{latin}
	
	هر سازنده (\lr{constructor}) این نوع استقرایی دقیقاً معادل یکی از قوانین استنتاجیِ ذکرشده در سند پروژه است (\lr{Assign}، \lr{Seq}، \lr{If-T}/\lr{If-F}، \lr{While-F}/\lr{While-T}). تنها تفاوت نحوی نسبت به مثال راهنمای \lr{determinism} این است که به‌جای \lr{forall x e s, ...}، از آرگومان‌های \textbf{ضمنی} (\lr{implicit}) با آکولاد \lr{\{\}} استفاده شده است؛ از نظر معنایی هیچ تفاوتی ندارد --- \lr{Lean} مقدار این آرگومان‌ها را خودش از روی زمینه (\lr{context}) استنتاج می‌کند --- و فقط کد را کوتاه‌تر و خواناتر می‌کند.
	
	نکته‌ی مهمی که در سازنده‌ی \lr{whileTrue} دیده می‌شود این است که خودش را در سومین مقدمه صدا می‌زند (\lr{BigStep (Stmt.while b body) sMid sFinal})؛ این بازگشتی‌بودن دقیقاً معنای «حلقه = یک بار اجرای بدنه + ادامه‌ی همان حلقه از حالت جدید» را می‌رساند و پایه‌ی اصلی روش اثبات قضیه در بخش بعد است.
	
	\section{قضیه‌ی انتخابی --- Loop Invariant}
	
	از میان سه گزینه‌ی صورت پروژه، \textbf{گزینه‌ی دوم (پایستگی ناوردای حلقه)} انتخاب و اثبات شده است.
	
	\subsection{صورت رسمی قضیه}
	\ltlbox{P(\sigma) \;\land\; \big(\forall\, \sigma_1\,\sigma_2,\; P(\sigma_1) \land \langle b,\sigma_1\rangle{\Downarrow}\text{dorost} \land \langle S,\sigma_1\rangle{\Downarrow}\sigma_2 \Rightarrow P(\sigma_2)\big) \;\land\; \langle \mathit{ta}(b)\,\{S\}, \sigma\rangle \Downarrow \sigma' \;\Rightarrow\; P(\sigma')}
	
	\paragraph{نکته‌ای درباره‌ی سور مقدمه‌ی دوم.}
	صورت پروژه مقدمه‌ی دوم را به شکل فشرده $P(\sigma) \land \langle S,\sigma\rangle{\Downarrow}\sigma'' \land P(\sigma'') \Rightarrow \cdots$ نوشته است، یعنی روی حالت‌های \emph{ثابت} $\sigma$ و $\sigma''$. در فرمول بالا، این مقدمه عمداً با یک سور همگانی $\forall\,\sigma_1\,\sigma_2$ بازنویسی شده است، به این دلیل که اثبات با استقرا روی تعداد دلخواه تکرار حلقه پیش می‌رود: در تکرار دوم به بعد، بدنه‌ی حلقه دیگر از همان $\sigma$ اولیه شروع نمی‌شود، بلکه از حالت میانیِ حاصل از تکرار قبلی آغاز می‌گردد. برای این‌که بتوان «حفظ‌شدن $P$ توسط بدنه» را در \emph{هر} یک از این تکرارها --- نه فقط تکرار اول --- به کار برد، این خاصیت باید برای تمام جفت‌حالت‌های ممکن برقرار فرض شود، نه فقط برای یک جفت خاص. همین سور همگانی است که در کد \lr{Lean} به‌صورت \lr{forall s s1, ...} در فرض \lr{bodyPreservesInvariant} ظاهر شده و امکان فراخوانی مکرر آن در هر لایه از استقرا (در شاخه‌ی \lr{whileTrue}) را فراهم می‌کند؛ بدون این تعمیم، اثبات تنها برای حلقه‌هایی با \textbf{دقیقاً یک بار} تکرار درست از آب درمی‌آمد. علاوه بر این، شرط $\langle b,\sigma_1\rangle{\Downarrow}\mathit{dorost}$ (که در نگارش فشرده‌ی صورت پروژه نیامده) نیز به‌صراحت اضافه شده تا مشخص باشد این پایستگی فقط وقتی لازم است که شرط حلقه واقعاً برقرار بوده و بدنه اجرا شده باشد.
	
	به زبان ساده: اگر
	\begin{enumerate}
		\item ناوردای $P$ قبل از شروع حلقه روی $\sigma$ برقرار باشد، و
		\item بدنه‌ی حلقه $S$، هر بار که با شرط برقرار ($b$ درست) از حالتی که $P$ در آن صادق است اجرا شود، $P$ را حفظ کند،
	\end{enumerate}
	آنگاه بعد از پایان \emph{کامل} حلقه‌ی \lr{ta}$(b)\{S\}$ (یعنی وقتی شرط سرانجام نادرست شد و حلقه با \lr{whileFalse} تمام شد)، $P$ روی حالت نهایی $\sigma'$ هم برقرار است.
	
	\subsection{چرا این قضیه؟}
	دلیل انتخاب این گزینه ارتباط مستقیم آن با خصوصیت \lr{Invariant} فاز دوم است. در فاز دوم، با \lr{SPIN} فقط نشان دادیم که در \emph{یک برنامه‌ی خاص} با یک حلقه‌ی مشخص، متغیر \lr{x} منفی نمی‌شود --- نتیجه‌ای که به‌درستی در گزارش فاز دوم «تا حدی بدیهی» توصیف شد، چون وابسته به مقادیر اولیه‌ی همان برنامه بود. قضیه‌ی \lr{loopInvariant} این محدودیت را برمی‌دارد: به‌جای بررسی یک برنامه با \lr{Model Checking}، ثابت می‌کند که \textbf{اگر} بدنه‌ی هر حلقه‌ای طوری نوشته شده باشد که یک شرط را حفظ کند، \textbf{آنگاه} بدون توجه به تعداد تکرار یا مقادیر اولیه، آن شرط بعد از حلقه هم برقرار می‌ماند. این دقیقاً همان قاعده‌ی استنتاج حلقه در منطق هُور است و پایه‌ی اثبات صحت هر برنامه‌ی حاوی \lr{while} به‌شمار می‌رود.
	
	\section{کد کامل اثبات و توضیح هر قدم}
	
	\begin{latin}
		\begin{lstlisting}[style=leanstyle]
			theorem loopInvariant
			{b : Expr}
			{body : Stmt}
			{P : State -> Prop}
			(bodyPreservesInvariant :
			forall s s1,
			P s ->
			evalExpr b s ≠ 0 ->
			BigStep body s s1 ->
			P s1)
			:
			forall s sFinal,
			P s ->
			BigStep (Stmt.while b body) s sFinal ->
			P sFinal := by
			
			intro s sFinal hP hRun
			
			have aux :
			forall {S : Stmt} {s sFinal : State},
			BigStep S s sFinal ->
			S = Stmt.while b body ->
			P s ->
			P sFinal := by
			
			intro S s0 s1 h
			induction h with
			
			| assign =>
			intro hEq
			cases hEq
			
			| seq hFirst hSecond ihFirst ihSecond =>
			intro hEq
			cases hEq
			
			| ifTrue hCond hThen ih =>
			intro hEq
			cases hEq
			
			| ifFalse hCond hElse ih =>
			intro hEq
			cases hEq
			
			| whileFalse hCond =>
			intro hEq hInv
			cases hEq
			exact hInv
			
			| whileTrue hCond hBody hRest ihBody ihRest =>
			intro hEq hInv
			cases hEq
			have hInvAfterBody : P _ :=
			bodyPreservesInvariant _ _ hInv hCond hBody
			exact ihRest rfl hInvAfterBody
			
			exact aux hRun rfl hP
		\end{lstlisting}
	\end{latin}
	
	\subsection{صورت‌بندی قضیه و ورود مقدمات}
	سه آرگومان ضمنی $b$، $body$ و $P$، و یک فرض اصلی به نام \lr{bodyPreservesInvariant} گرفته می‌شوند که دقیقاً مقدمه‌ی شماره‌ی ۲ از صورت ریاضی بالا را بیان می‌کند: «برای هر دو حالت $s$ و $s_1$، اگر $P(s)$ برقرار باشد، شرط حلقه روی $s$ درست باشد، و بدنه از $s$ به $s_1$ برسد، آنگاه $P(s_1)$.» خط \lr{intro s sFinal hP hRun} هم بقیه‌ی مقدمات (حالت اولیه، حالت نهایی، برقراری $P$ در ابتدا، و خودِ اشتقاق اجرای حلقه) را وارد می‌کند.
	
	\subsection{چرا یک لم کمکی (aux) لازم است؟}
	اینجا مهم‌ترین تفاوت این اثبات با مثال راهنمای \lr{determinism} است. در مثال راهنما، فرض \lr{h1 : BigStep S s s'} روی یک متغیر \textbf{عمومی} $S$ بود، پس می‌شد مستقیماً \lr{induction h1} زد. اما اینجا فرض \lr{hRun} روی یک دستور \textbf{خاص و ثابت}، یعنی دقیقاً \lr{Stmt.while b body}، است. اگر بخواهیم مستقیم روی \lr{hRun} استقرا بزنیم، \lr{Lean} برای هر یک از شش سازنده‌ی \lr{BigStep} (نه فقط \lr{whileFalse} و \lr{whileTrue}) یک شاخه می‌سازد، اما نمی‌تواند به‌طور خودکار متوجه شود که آرگومان اول در چهار سازنده‌ی دیگر (\lr{assign}، \lr{seq}، \lr{ifTrue}، \lr{ifFalse}) هرگز نمی‌تواند برابر \lr{Stmt.while b body} باشد.
	
	راه‌حل استاندارد (که در اثبات‌های \lr{Big-Step} در \lr{Lean} و \lr{Coq} بسیار رایج است) این است که به‌جای استقرا مستقیم روی \lr{hRun}، یک گزاره‌ی کمکی \lr{aux} تعریف می‌کنیم که روی یک دستور \textbf{عمومی} \lr{S} بیان شده، به‌همراه یک فرض تساوی جداگانه \lr{S = Stmt.while b body}. حالا می‌توان روی اشتقاق \lr{h : BigStep S s0 s1} به‌طور کاملاً عمومی استقرا زد (چون \lr{S} دیگر ثابت نیست)، و در هر شاخه، فرض تساوی را با \lr{cases hEq} باز کرد:
	\begin{itemize}
		\item در چهار شاخه‌ی \lr{assign}، \lr{seq}، \lr{ifTrue} و \lr{ifFalse}، فرض \lr{hEq} چیزی از این جنس است: \lr{Stmt.assign x e = Stmt.while b body}. چون این دو طرف تساوی از دو سازنده‌ی \textbf{متفاوت} یک نوع استقرایی ساخته شده‌اند، چنین تساوی‌ای اساساً \textbf{غیرممکن} است. تاکتیک \lr{cases hEq} این تناقض را تشخیص می‌دهد و بلافاصله آن شاخه را می‌بندد --- به همین دلیل بعد از \lr{cases hEq} در این چهار مورد هیچ خط دیگری لازم نیست؛ هدف با صفر اثبات باقی‌مانده بسته می‌شود.
		
		\item در شاخه‌ی \lr{whileFalse}، \lr{hEq} از نوع \lr{Stmt.while b' body' = Stmt.while b body} است. اینجا \lr{cases hEq} تساوی \lr{b' = b} و \lr{body' = body} را استخراج و در کل هدف جایگزین می‌کند. طبق تعریف سازنده‌ی \lr{whileFalse}، حالت پایانی همان حالت شروع است (\lr{s1 = s0})، پس هدف باقی‌مانده دقیقاً \lr{P s0} می‌شود که همان \lr{hInv} است.
		
		\item در شاخه‌ی \lr{whileTrue}، بعد از \lr{cases hEq} همان جایگزینی رخ می‌دهد. حالا از \lr{bodyPreservesInvariant} (با سه آرگومان \lr{hInv}، \lr{hCond} و \lr{hBody}) نتیجه می‌گیریم که ناوردا بعد از \textbf{یک بار} اجرای بدنه هم برقرار است: \lr{hInvAfterBody : P sMid}. سپس فرض استقرای مربوط به «ادامه‌ی حلقه» یعنی \lr{ihRest} را با دو آرگومان 
		\lr{rfl} (اثبات بدیهی تساوی \lr{Stmt.while b body = Stmt.while b body}) و \lr{hInvAfterBody} صدا می‌زنیم تا \lr{P sFinal} نتیجه شود.
		
	\end{itemize}
	
	\subsection{اتصال نهایی}
	در پایان، قضیه‌ی اصلی \lr{loopInvariant} تنها با یک خط \lr{exact aux hRun rfl hP} بسته می‌شود: \lr{hRun} همان اشتقاق اجرای حلقه است، \lr{rfl} گواهی بدیهی بر تساوی \lr{Stmt.while b body = Stmt.while b body}، و \lr{hP} فرض برقراری ناوردا در ابتدای حلقه.
	
	\subsection{یک نکته‌ی فنی درباره‌ی کامپایل}
	دو دستور \lr{\#check BigStep} و \lr{\#check loopInvariant} --- که صرفاً برای تأیید صحت \lr{type} تعاریف و بدون اثر روی خودِ اثبات هستند --- باید \textbf{بیرون} از بلوک \lr{by} قضیه (یعنی بعد از خط \lr{exact aux hRun rfl hP} و با تورفتگیِ سطح فایل) نوشته شوند، نه داخل بدنه‌ی تاکتیکی. در کد نهایی تحویلی، این دو خط به همین صورت اصلاح و در انتهای فایل، خارج از قضیه، قرار داده شده‌اند.
	
	\section{مثال گام‌به‌گام (شفاف‌سازی رفتار استقرا)}
	
	برای ملموس‌تر شدن مکانیزم بالا، اجرای یک حلقه‌ی فرضی با دقیقاً دو بار تکرار بدنه را در نظر بگیرید. اشتقاق \lr{hRun} به این شکل درخت می‌شود:
	
	\[
	\dfrac{
		\langle S,\sigma_0\rangle \Downarrow \sigma_1
		\quad
		\dfrac{
			\langle S,\sigma_1\rangle \Downarrow \sigma_2
			\quad
			\dfrac{
				\text{(شرط نادرست)}
			}{
				\langle \mathit{ta}(b)\{S\}, \sigma_2\rangle \Downarrow \sigma_2
			}\;\text{(whileFalse)}
		}{
			\langle \mathit{ta}(b)\{S\}, \sigma_1\rangle \Downarrow \sigma_2
		}\;\text{(whileTrue)}
	}{
		\langle \mathit{ta}(b)\{S\}, \sigma_0\rangle \Downarrow \sigma_2
	}\;\text{(whileTrue)}
	\]
	
استقرا روی این درخت دقیقاً از پایین (بیرونی‌ترین \lr{whileTrue}) شروع می‌شود: ابتدا با \lr{bodyPreservesInvariant} از $P(\sigma_0)$ به $P(\sigma_1)$ می‌رسیم، سپس \lr{ihRest} روی زیر‌اشتقاق داخلی فراخوانی می‌شود که خودش دوباره یک \lr{whileTrue} است --- پس دوباره با \lr{bodyPreservesInvariant} از $P(\sigma_1)$ به $P(\sigma_2)$ می‌رسیم، و در نهایت با \lr{whileFalse} در عمیق‌ترین لایه، $P(\sigma_2)$ بدون تغییر به عنوان نتیجه‌ی نهایی برگردانده می‌شود. یعنی اثبات به تعداد تکرارهای واقعی حلقه، «باز» می‌شود --- دقیقاً همان چیزی که استقرای ساختاری روی \lr{BigStep} تضمین می‌کند، بدون آنکه لازم باشد تعداد تکرارها را از پیش بدانیم یا در قضیه ذکر کنیم.

	\section{قضیه‌ی دوم --- Expression Determinism}
	
	علاوه بر قضیه‌ی \lr{Loop Invariant}، گزینه‌ی اول صورت پروژه --- قطعی‌بودن ارزیابی عبارات --- نیز در فایل \lr{exprDeterminism.lean} اثبات شده است.
	
	\subsection{چرا این قضیه به یک تعریف جدید نیاز داشت}
	
	در \lr{Basic.lean}، تابع \lr{evalExpr} به‌صورت یک \textbf{تابع بازگشتی معمولی} تعریف شده، نه یک رابطه‌ی استقرایی. طبق تعریف خودِ مفهوم تابع در \lr{Lean}، هر تابع برای هر ورودی دقیقاً یک خروجی دارد؛ پس قطعی‌بودنِ این نسخه از ارزیابی یک ادعای بدیهی و بی‌محتواست (اثباتش فقط \lr{rfl} است) و چیزی درباره‌ی درستی معناشناسی زبان به ما نمی‌گوید.
	
	برای این‌که این قضیه معنای واقعی خودش --- همان چیزی که در سند پروژه با نماد $\langle e,\sigma\rangle \Downarrow v$ بیان شده --- را داشته باشد، باید ارزیابی عبارات را هم، درست مثل \lr{BigStep}، به‌صورت یک \textbf{رابطه‌ی استقرایی} جداگانه تعریف کرد و قطعی‌بودن \emph{همان رابطه} را اثبات کرد. این دقیقاً همان کاری است که در مثال راهنمای پروژه برای \lr{BigStep} انجام شده بود، با این تفاوت که این‌جا روی \lr{Expr} به‌جای \lr{Stmt} تکرار می‌شود.
	
	\subsection{تعریف رابطه‌ی EvalExpr}
	
	\begin{latin}
		\begin{lstlisting}[style=leanstyle]
			inductive EvalExpr : Expr -> State -> Int -> Prop
			| num : forall n s, EvalExpr (Expr.num n) s n
			| var : forall x s, EvalExpr (Expr.var x) s (s x)
			| add : forall e1 e2 s v1 v2,
			EvalExpr e1 s v1 -> EvalExpr e2 s v2 ->
			EvalExpr (Expr.add e1 e2) s (v1 + v2)
			| lt : forall e1 e2 s v1 v2,
			EvalExpr e1 s v1 -> EvalExpr e2 s v2 ->
			EvalExpr (Expr.lt e1 e2) s (if v1 < v2 then 1 else 0)
		\end{lstlisting}
	\end{latin}
	
	هر سازنده‌ی این نوع دقیقاً معادل یکی از قوانین استنتاجی بخش «ارزیابی عبارات» در سند پروژه است: \lr{num} معادل قانون \lr{(Num)}، \lr{var} معادل \lr{(Var)}، \lr{add} معادل \lr{(Add)}، و \lr{lt} معادل \lr{(Lt)}. تفاوت این تعریف با تابع \lr{evalExpr}، این است که اینجا هیچ‌چیز از پیش تضمین نمی‌کند یک عبارت مشخص فقط به یک مقدار برسد --- ممکن است (تا پیش از اثبات قضیه) دو اشتقاق متفاوت برای \lr{EvalExpr} روی یک عبارت و یک حالت وجود داشته باشد که به دو مقدار متفاوت برسند؛ همین امکانِ (نظری) عدم‌قطعیت است که قضیه‌ی زیر آن را رد می‌کند.
	
	\subsection{بیان و اثبات قضیه}
	
	\ltlbox{\langle e,\sigma\rangle \Downarrow v_1 \;\land\; \langle e,\sigma\rangle \Downarrow v_2 \;\Rightarrow\; v_1 = v_2}
	
	از آن‌جا که در این صورت‌بندی، برخلاف قضیه‌ی \lr{loopInvariant}، عبارت $e$ از ابتدا یک متغیر عمومی است (نه ثابت به شکلی خاص مثل \lr{Stmt.while b body})، نیازی به ترفند \lr{remember}/تساوی کمکی نیست؛ می‌توان مستقیماً روی اشتقاق فرض اول (\lr{h1}) استقرا زد، دقیقاً مثل اثبات \lr{determinism} در مثال راهنما. تنها نکته این است که چون فرض دوم (\lr{h2}) روی متغیر آزاد \lr{v2} است، باید با \lr{generalizing v2} این متغیر را هم در هر گام استقرا آزاد نگه داشت، وگرنه فرض استقرا برای مقدار درستی از \lr{v2} در دسترس نخواهد بود.
	
	منطق هر شاخه:
	\begin{itemize}
		\item \textbf{num}: چون $e$ به شکل \lr{Expr.num n} ثابت است، تنها قانونی که می‌تواند \lr{h2} را هم تولید کرده باشد \lr{(Num)} است؛ تاکتیک \lr{cases h2} این را تشخیص می‌دهد و بلافاصله هر دو مقدار را برابر با خودِ $n$ قرار می‌دهد، پس هدف با \lr{rfl} بسته می‌شود.
		\item \textbf{var}: با همان استدلال، تنها قانون ممکن برای \lr{h2} روی \lr{Expr.var x}، قانون \lr{(Var)} است؛ هر دو مقدار برابر $\sigma(x)$ می‌شوند.
		\item \textbf{add}: \lr{cases h2} نشان می‌دهد اشتقاق دوم هم باید از قانون \lr{(Add)} استفاده کرده باشد (چون $e$ به شکل \lr{add e1 e2} ثابت است)، و دو زیراشتقاق \lr{h2a}, \lr{h2b} برای $e_1$ و $e_2$ به دست می‌آید. با اعمال فرض‌های استقرای \lr{iha} و \lr{ihb} روی این دو زیراشتقاق، برابری مقادیر $e_1$ و $e_2$ در دو اجرا نتیجه می‌شود؛ با \lr{rw} این دو تساوی را در هدف جایگزین می‌کنیم تا $v_1+v_2 = v_1'+v_2'$ به‌صورت خودکار بسته شود.
		\item \textbf{lt}: کاملاً مشابه \lr{add}، با این تفاوت که در پایان به‌جای جمع، همان عبارت شرطی \lr{if v1 < v2 then 1 else 0} با مقادیر یکسان جایگزین و برابری نتیجه می‌شود.
	\end{itemize}
	
	در هیچ‌کدام از این چهار شاخه امکان تناقض بین دو قانون مختلف مطرح نیست (برخلاف قضیه‌ی \lr{determinism} روی \lr{BigStep} در مثال راهنما، که در آن شاخه‌های \lr{ifTrue}/\lr{ifFalse} و \lr{whileF}/\lr{whileT} می‌توانستند با هم تناقض کنند)، چون \lr{Expr} هیچ سازنده‌ی شرطی (مثل \lr{if} یا \lr{while}) ندارد؛ هر سازنده‌ی \lr{Expr} دقیقاً یک قانون \lr{EvalExpr} متناظر دارد، پس \lr{cases h2} در هر شاخه بلافاصله یک نتیجه‌ی یکتا می‌دهد.
	
	\subsection{کد کامل}
	
	\begin{latin}
		\begin{lstlisting}[style=leanstyle]
			theorem exprDeterminism : forall e s v1 v2,
			EvalExpr e s v1 -> EvalExpr e s v2 -> v1 = v2 := by
			intro e s v1 v2 h1 h2
			induction h1 generalizing v2 with
			| num n s =>
			cases h2
			rfl
			| var x s =>
			cases h2
			rfl
			| add e1 e2 s va vb _ _ iha ihb =>
			cases h2 with
			| add h2a h2b =>
			have ea := iha _ h2a
			have eb := ihb _ h2b
			rw [ea, eb]
			| lt e1 e2 s va vb _ _ iha ihb =>
			cases h2 with
			| lt h2a h2b =>
			have ea := iha _ h2a
			have eb := ihb _ h2b
			rw [ea, eb]
			
			#check exprDeterminism
		\end{lstlisting}
	\end{latin}
	
	\subsection{جمع‌بندی}
	
	این قضیه نشان می‌دهد که حتی اگر ارزیابی عبارات را --- برخلاف تعریف تابعی \lr{evalExpr} --- به‌صورت یک رابطه‌ی استقرایی آزاد بیان کنیم، باز هم برای هر عبارت و هر حالت، حداکثر یک مقدار قابل اشتقاق است. این نتیجه مکمل قضیه‌ی \lr{determinism} روی \lr{Stmt}/\lr{BigStep} (مثال راهنمای پروژه) است: در کنار هم، این دو قضیه تضمین می‌کنند که کل معناشناسی \lr{Hash} --- هم روی عبارات و هم روی دستورات --- کاملاً قطعی و بدون ابهام تعریف شده است.
	
	\section{جمع‌بندی فاز سوم}
	
	در این فاز، با تعریف صوری \lr{Big-Step} برای زیرمجموعه‌ی \lr{Hash} و اثبات قضیه‌ی پایستگی ناوردای حلقه در \lr{Lean 4}، از سطح بررسی موردیِ فاز دوم (که فقط درباره‌ی یک برنامه‌ی خاص با \lr{SPIN} صحبت می‌کرد) به یک تضمین ریاضی و کلی رسیدیم که برای \textbf{هر} برنامه‌ی \lr{Hash} حاوی یک حلقه‌ی \lr{ta} با بدنه‌ی ناوردا-حفظ‌کننده صادق است.
